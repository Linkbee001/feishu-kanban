    import { INestApplication, ValidationPipe } from '@nestjs/common';
    import { AppModule } from '../../src/app.module';
    import { AdminAuthGuard } from '../../src/common/auth/admin-auth.guard';
    import { HttpExceptionFilter } from '../../src/common/errors/http-exception.filter';

    export async function createE2eTestModule(): Promise<INestApplication> {
      const moduleFixture: TestingModule = await Test.createTestingModule({
        imports: [AppModule],
      })
        .overrideGuard(AdminAuthGuard)
        .useValue({ canActivate: () => true })
        .compile();

      const app = moduleFixture.createNestApplication();
      app.useGlobalPipes(new ValidationPipe({ whitelist: true, transform: true }));
      app.useGlobalFilters(new HttpExceptionFilter());
      await app.init();
      return app;
    }
    ```

    export class E2eTestFixture {
      private prisma: PrismaService;
      private testChatIds: string[] = [];

      async setup(prisma: PrismaService) {
        this.prisma = prisma;
      }

      createTestChatId(): string {
        const chatId = `oc_e2e_test_${Date.now()}_${Math.random().toString(36).slice(2)}`;
        this.testChatIds.push(chatId);
        return chatId;
      }

      async cleanup() {
        for (const chatId of this.testChatIds) {
          await this.cleanupByChatId(chatId);
        }
      }

      private async cleanupByChatId(chatId: string) {
        // Delete in correct order (respecting foreign key constraints)
        await this.prisma.runtimeEvent.deleteMany({ where: { chatId } });
        await this.prisma.agentRun.deleteMany({ where: { chatId } });
        await this.prisma.messageSource.deleteMany({ where: { feishuChatId: chatId } });
        await this.prisma.groupAgentSession.deleteMany({ where: { feishuChatId: chatId } });
        await this.prisma.project.deleteMany({ where: { feishuChatId: chatId } });
      }
    }
    ```
