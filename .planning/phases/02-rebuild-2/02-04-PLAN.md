        return;

      // If session is somehow in bootstrap mode (legacy), still handle gracefully
      if (session.sessionMode === GroupSessionMode.bootstrap) {
        if (isBotMentioned) {
          await this.feishu.sendTextMessage(

            chatId,
            '本群正在使用旧的初始化流程，请联系管理员迁移到新的配置管理模式。',
